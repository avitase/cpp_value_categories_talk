#include <memory>

struct Resource {};

struct Target {
    Target(const Resource&) { std::cout << 'a'; }
    Target(Resource&&) { std::cout << 'b'; }
};

    return std::make_unique<Target>(???);
}

int main() {
    Resource resource;
    make_target(resource);            // should print 'a'
    make_target(Resource(resource));  // should print 'b'

#include <cstddef>
#include <type_traits>
#include <utility>

template <typename T>
struct Data final {
    T *data;
    explicit Data(const std::size_t size): data(new T[size]) {}
    ~Data() { delete [] data; }
};

auto init() {
    Data<int> d(3);
    d.data[0] = 1; d.data[1] = 2; d.data[2] = 3;
    return d;
}

int main() { return init().data[2]; }
